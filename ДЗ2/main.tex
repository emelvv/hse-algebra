    $$G = \mat{* & * \\0 & *}, \quad H = \mat{1 & * \\0 & *}$$
    Покажем, что:
    $$g^{-1}Hg = H \quad \forall g \in G$$
    Пусть:
    $$g \in G, h \in H \implies g = \mat{a & b\\0 & c}, h = \mat{1 & d \\ 0 & e}$$
    Тогда:
    $$g^{-1}Hg = \mat{a & b\\0 & c} \cdot \mat{1 & d \\ 0 & e} \cdot  \mat{a & b\\0 & c}  = \begin{pmatrix}
        a^2 & ab + c(ad + be) \\
        0 & c^2e
        \end{pmatrix}  \in H$$
    \textbf{Следовательно, $H \triangle G$}
    

\end{enumerate}
